=== Wireless Weightlifting Trolley - Hardware & Software Guide ===

1. COMPLETE PARTS LIST (FINAL PROTOTYPE CONFIGURATION)
----------------------------------------------------
- 1x ESP32 DevKit V1 (30-pin version)
- 1x 4-Channel 5V Relay Module (Existing)
- 1x 2-Channel 5V Relay Module (New addition)
- 3x 5840-31ZY 24V Worm Gear Motors (1x Lift, 2x Navigation)
- 1x 24V 15A 360W SMPS (Switching Power Supply)
- 1x LM2596 DC-DC Buck Converter (To step down 24V to 5.0V)
- 1x 15A Blade Fuse + Inline Fuse Holder
- 1x Heavy Duty Toggle Switch (Master ON/OFF)
- 10 Meters 16 AWG Power Wire (Red/Black) for 24V connections
- 40x Jumper Wires (F-F, M-F, M-M) for 5V logic connections
- 1x Breadboard (830 point) for ground and 5V power distribution
- 2x Drive Wheels for navigation motors
- 1x Caster Wheel (Rear Support)
- 1x Digital Multimeter (For 5.0V calibration)

2. POWER DISTRIBUTION (WIRING STRATEGY)
--------------------------------------
A. High Power Loop (24V Line):
- SMPS 24V+ -> Fuse -> Toggle Switch -> Relay 'Normally Open' (NO) terminals on BOTH modules.
- SMPS GND -> Relay 'Normally Closed' (NC) terminals on BOTH modules.
- Lift Motor: 4-Channel Module (Relay 1 & 2 COM).
- Left Motor: 4-Channel Module (Relay 3 & 4 COM).
- Right Motor: 2-Channel Module (Relay 1 & 2 COM).

B. Logic Power Loop (5V Line):
- SMPS 24V+ -> LM2596 IN+
- SMPS GND  -> LM2596 IN-
- LM2596 OUT+ (Tuned to 5.0V exactly) -> ESP32 VIN, 4-Ch Module VCC, and 2-Ch Module VCC.
- LM2596 OUT- -> Common GND for all modules.

3. ESP32 PIN CONNECTIONS
------------------------
- 4-Channel Module:
  - IN1 (Lift Up): GPIO 12
  - IN2 (Lift Down): GPIO 13
  - IN3 (Left Fwd): GPIO 14
  - IN4 (Left Bwd): GPIO 27
- 2-Channel Module:
  - IN1 (Right Fwd): GPIO 26
  - IN2 (Right Bwd): GPIO 25

#include <WiFi.h>
#include <WebServer.h>

const int liftUpPin = 12; const int liftDownPin = 13;
const int leftFwdPin = 14; const int leftBwdPin = 27;
const int rightFwdPin = 26; const int rightBwdPin = 25;

WebServer server(80);

void handleRoot() { server.send(200, "text/html", htmlPage); }

// Relay Active Low Logic
void stopLift() { digitalWrite(liftUpPin, HIGH); digitalWrite(liftDownPin, HIGH); server.send(200); }
void stopNav() {
  digitalWrite(leftFwdPin, HIGH); digitalWrite(leftBwdPin, HIGH);
  digitalWrite(rightFwdPin, HIGH); digitalWrite(rightBwdPin, HIGH);
  server.send(200);
}

void liftUp() { digitalWrite(liftUpPin, LOW); digitalWrite(liftDownPin, HIGH); server.send(200); }
void liftDown() { digitalWrite(liftUpPin, HIGH); digitalWrite(liftDownPin, LOW); server.send(200); }
void forward() {
  digitalWrite(leftFwdPin, LOW); digitalWrite(leftBwdPin, HIGH);
  digitalWrite(rightFwdPin, LOW); digitalWrite(rightBwdPin, HIGH);
  server.send(200);
}
void backward() {
  digitalWrite(leftFwdPin, HIGH); digitalWrite(leftBwdPin, LOW);
  digitalWrite(rightFwdPin, HIGH); digitalWrite(rightBwdPin, LOW);
  server.send(200);
}
void turnLeft() {
  digitalWrite(leftFwdPin, HIGH); digitalWrite(leftBwdPin, LOW);
  digitalWrite(rightFwdPin, LOW); digitalWrite(rightBwdPin, HIGH);
  server.send(200);
}
void turnRight() {
  digitalWrite(leftFwdPin, LOW); digitalWrite(leftBwdPin, HIGH);
  digitalWrite(rightFwdPin, HIGH); digitalWrite(rightBwdPin, LOW);
  server.send(200);
}

void setup() {
  int pins[] = {12, 13, 14, 27, 26, 25};
  for(int p : pins) { pinMode(p, OUTPUT); digitalWrite(p, HIGH); }
  WiFi.softAP("Trolley_Pro", "12345678");
  server.on("/", handleRoot);
  server.on("/liftup", liftUp); server.on("/liftdown", liftDown); server.on("/stoplift", stopLift);
  server.on("/fwd", forward); server.on("/bwd", backward);
  server.on("/left", turnLeft); server.on("/right", turnRight);
  server.on("/stopnav", stopNav); server.on("/stopall", [](){ stopLift(); stopNav(); });
  server.begin();
}

void loop() { server.handleClient(); }
